#include <stdio.h>
int main()
{
    int n, i, j, temp;
    int arr[10];
 
    printf("How many elements do you wish to enter?\n");
    scanf("%d", &n);
 
    printf("\nWhat are your %d elements of choice?\n", n);
    for (i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }
    for (i = 1 ; i <= n - 1; i++)
    {
	    j = i;
            while ( j > 0 && arr[j-1] > arr[j])
            {	        
                temp     = arr[j];
                arr[j]   = arr[j-1];
                arr[j-1] = temp;
                j--;
            }
    }
    printf("\nThe list after insertion sorting: \n");
    for (i = 0; i <= n - 1; i++)
    {
        printf("%d\n", arr[i]);
    }
    return 0;
}
